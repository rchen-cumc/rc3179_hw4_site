data(rest_inspec)

rest_inspec =
  rest_inspec %>%
  filter(grade != 'Not Yet Graded', boro != "Missing") %>% 
  mutate(boro = str_to_title(boro))

Restaurant Inspection Grades by Borough

rest_inspec %>% 
  mutate(boro = as.factor(boro)) %>% 
  group_by(boro, grade) %>% 
  summarize(n = n()) %>% 
  plot_ly(x = ~boro, y = ~n, color = ~grade, type = "bar")
